Exercise 3: Write a function printNumbers(from, to) that outputs a number every second, starting from from and ending with to.

                    const printNumbers = function (from, to) {
                        let base=1;
                        while(from<=to){
                            setTimeout(function(x){
                                console.log(x);
                            },base*1000,from);
                            from++;
                            base++;
                        }
                        return;
                    };
